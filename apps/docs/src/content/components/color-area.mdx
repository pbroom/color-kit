# Color Area

`ColorArea` is a two-dimensional control for channel pairs like chroma/lightness.

<ComponentPreview>
  <ColorAreaDemo />
</ComponentPreview>

## API

<ApiTable rows={componentApiDocs.colorArea} />

## Usage

```tsx
import {
  Background,
  ColorArea,
  ColorPlane,
  ContrastRegionLayer,
  FallbackPointsLayer,
  GamutBoundaryLayer,
} from '@color-kit/react';

<ColorArea axes={{ x: { channel: 'c' }, y: { channel: 'l' } }}>
  <Background checkerboard />
  <ColorPlane />
  <GamutBoundaryLayer gamut="display-p3" />
  <ContrastRegionLayer threshold={4.5} />
  <FallbackPointsLayer />
</ColorArea>;
```
